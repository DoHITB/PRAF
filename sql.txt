DROP TABLE IF EXISTS `PARDATA`;
CREATE TABLE IF NOT EXISTS `PARDATA` (
  `PARID` int(20) NOT NULL AUTO_INCREMENT,
  `PARPIV` longtext COLLATE latin1_general_ci NOT NULL,
  `PARPAT` longtext COLLATE latin1_general_ci NOT NULL,
  `PARISMORE` int(20) NOT NULL,
  PRIMARY KEY (`PARID`)
) ENGINE=MyISAM  DEFAULT CHARSET=latin1 COLLATE=latin1_general_ci AUTO_INCREMENT=1003 ;
